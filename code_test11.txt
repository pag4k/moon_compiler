// DEEPLY NESTED FUNCTION CALL AND RETURNING ARRAY

class a {
  b b1;
  b b2;
  b b3;
};

class b {
  c c1;
  c c2;
  c c3;
};

class c {
  d[4] fillArray(integer n);
};

class d {
  integer d;
};

d[4] c::fillArray(integer n) {
  d d[4];
  for (integer i = 0; i < 4; i = i +1) {
    d[i].d = n;
  };
  return(d);
};


main {
  a a;
  d d[4];
  d = a.b2.c2.fillArray(6);
  for(integer i = 0; i < 4; i = i + 1) {
    write(d[i].d);
  };
};
