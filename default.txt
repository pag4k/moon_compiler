class pos {
  integer x;
  integer y;
  integer z;
  
};

pos new(integer x, integer y, integer z) {
  pos new;
  new.x = x;
  new.y = y;
  new.z = z;
  return(new);
};

pos add(pos a1, pos a2) {
  a1.x = a1.x + a2.x;
  a1.y = a1.y + a2.y;
  a1.z = a1.z + a2.z;
  return(a1);
};


main {

  pos a;
  a = new(1,2,3);
  pos b;
  b = new(10,20,30);
  a = add(a,b);
  write(a.x);
  write(a.y);
  write(a.z);

  
};
