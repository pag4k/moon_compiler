// NESTER FOR AND MULTIDIMENSIONAL ARRAYS

class a {
  integer a;
  integer b;
};

main {
  a a[10][5];
  for(integer i = 0; i < 10; i = i + 1) {
    for(integer j = 4; j >= 0; j = j - 1) {
      a[i][j].a = i * j;
      a[i][j].b = -1;
    };
  };
  
  for(integer k = 0; k < 10; k = k + 1)
    for(integer l = 4; l >= 0; l = l - 1)
      write(a[k][l].a);
    ;
  ;

  for(integer m = 0; m < 10; m = m + 1)
    for(integer n = 4; n >= 0; n = n - 1)
      write(a[m][n].b);
    ;
  ;

};
